async function submitFlag() { const flag = document.getElementById('flagInput').value; const resEl = document.getElementById('responseMessage'); try { const response = await fetch('/api/round2/submit', { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-Team-ID': localStorage.getItem('teamId') }, body: JSON.stringify({ flag }) }); const result = await response.json(); if (result.state && result.state.isCorrect) { resEl.textContent = 'Correct'; resEl.className = 'success'; } else { resEl.textContent = 'Wrong'; resEl.className = 'error'; } } catch (err) { resEl.textContent = 'Error submitting flag. Try again.'; resEl.className = 'error'; } }
